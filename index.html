2. Program wczytuje liczbe (dzien tygodnia) i wypisuje dzien zależnie od tego jaka liczbe wpiszemy. _old

 #!/bin/bash
 
#!/bin/bash
echo "Podaj numer dnia tygodnia: "
read dzien
case "$dzien" in
  "1") echo "poniedzialek" ;;
  "2") echo "wtorek" ;;
  "3") echo "sroda" ;;
  "4") echo "czwartek" ;;
  "5") echo "piatek" ;;
  "6") echo "sobota" ;;
  "7") echo "niedziela" ;;
  *) echo "Nie rozumiem"
esac
exit 0
